022package com.nexmo.client.auth;
023
024import com.nexmo.client.NexmoClientException;
025
026import java.util.Set;
027import java.util.SortedSet;
028import java.util.TreeSet;
029
030/**
031 * Internal class, managing a collection of {@link AuthMethod}s.
032 * <p>
033 * This holds a collection of AuthMethod instances, in order of preference, and
034 * allow for simple selection of an appropriate AuthMethod for a particular REST endpoint.
035 */
036public class AuthCollection {
037    private SortedSet<AuthMethod> authList;
038
039    /**
040     * Create a new AuthCollection with an empty set of AuthMethods.
041     */
042    public AuthCollection() {
043        this.authList = new TreeSet<>();
044    }
045
046    public AuthCollection(AuthMethod... authMethods) {
047        this();
048        for (AuthMethod method : authMethods) {
049            add(method);
050        }
051    }
052
053    /**
054     * Add a new {@link AuthMethod} to the set managed by this AuthCollection
055     *
056     * @param auth AuthMethod method to be added to this collection
057     */
058    public void add(AuthMethod auth) {
059        this.authList.add(auth);
060    }
061
062    /**
063     * Obtain an {@link AuthMethod} instance for a set of acceptable AuthMethod classes.
064     *
065     * @param acceptableAuthMethodClasses A Set of AuthMethod classes which are suitable for the target REST endpoint.
066     * @return the preferred AuthMethod from the provided set of acceptable AuthMethod classes
067     * @throws NexmoUnacceptableAuthException if no appropriate AuthMethod is held by this AuthCollection
068     */
069    public AuthMethod getAcceptableAuthMethod(Set<Class> acceptableAuthMethodClasses) throws NexmoClientException {
070        for (AuthMethod availableAuthMethod : this.authList) {
071            if (acceptableAuthMethodClasses.contains(availableAuthMethod.getClass())) {
072                return availableAuthMethod;
073            }
074        }
075        throw new NexmoUnacceptableAuthException(this.authList, acceptableAuthMethodClasses);
076    }
077}
